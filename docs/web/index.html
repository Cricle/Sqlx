🗂️ 通用CRUD接口 - 零样板代码

继承 ICrudRepository<TEntity, TKey> 立即获得8个标准CRUD方法，无需手写SQL！ 

// 1️⃣ 定义实体
public class Product
{
    public int Id { get; init; }
    public string Name { get; init; } = "";
    public decimal Price { get; init; }
    public DateTime CreatedAt { get; init; }
}

// 2️⃣ 继承通用接口（一行代码搞定！）
[RepositoryFor<Product>]
[TableName("products")]
public partial class ProductRepository : ICrudRepository<Product, int>
{
    public ProductRepository(DbConnection connection) { }
}

// 3️⃣ 使用（8个方法全自动生成 ✨）
var product = await repo.GetByIdAsync(1);           // 根据ID查询
var all = await repo.GetAllAsync(limit: 10);        // 分页查询
await repo.InsertAsync(newProduct);                 // 插入
await repo.UpdateAsync(product);                    // 更新
await repo.DeleteAsync(1);                          // 删除
var count = await repo.CountAsync();                // 统计总数
var exists = await repo.ExistsAsync(1);             // 检查存在
await repo.BatchInsertAsync(products);              // 批量插入

✅ 生成的SQL（最佳实践）

-- GetByIdAsync
SELECT id, name, price, created_at FROM products WHERE id = @id

-- GetAllAsync

ORDER BY id LIMIT @limit OFFSET @offset

INSERT INTO products (name, price, created_at) VALUES

  (@name_2, @price_2, @created_at_2)

1. 定义实体和接口

// 实体类
public class User
{
    public int Id { get; set; }
    public string Name { get; set; } = string.Empty;
    public string Email { get; set; } = string.Empty;
    public DateTime? LastLogin { get; set; }  // ✅ Nullable支持
}

// 数据访问接口
[Sqlx("SELECT * FROM users WHERE id = @id")]
public partial interface IUserService
{
    Task<User?> GetUserByIdAsync(int id);
}

2. 使用

var user = await userService.GetUserByIdAsync(123);
Console.WriteLine($"User: {user?.Name}");

正则表达式列筛选

// 只查询 user_ 开头的列
[Sqlx("SELECT {{columns --regex ^user_}} FROM users")]
List<User> GetUserColumnsOnly();

// 生成的SQL: SELECT user_name, user_email FROM users

// 排除敏感字段

动态返回值（运行时不确定的列）

// 返回Dictionary，适用于报表查询
[Sqlx("SELECT {{columns --regex @pattern}} FROM {{@tableName}}")]
List<Dictionary<string, object>> GetDynamicReport(
    [DynamicSql] string tableName,  // 动态表名
    string pattern);                // 正则模式

// 使用
var report = GetDynamicReport("users", "^user_");
foreach (var row in report)
{
    foreach (var (key, value) in row)
    {
        Console.WriteLine($"{key}: {value}");
    }
}

批量操作（高性能）

// 批量插入
[Sqlx("INSERT INTO users (name, email) VALUES {{batch_values}}")]
int BatchInsert([BatchOperation] List<User> users);

// 一次性插入1000行，比循环插入快10倍以上
var users = Enumerable.Range(1, 1000)
    .Select(i => new User { Name = $"User{i}", Email = $"user{i}@example.com" })
    .ToList();

int affected = await userService.BatchInsert(users);
Console.WriteLine($"插入了 {affected} 行");

Activity追踪和自定义拦截

// 自动生成Activity，集成OpenTelemetry

public partial class UserService
{
    // 可选的Partial方法
    partial void OnExecuting(string operationName, DbCommand command)
    {
        Console.WriteLine($"执行: {operationName}");
        Console.WriteLine($"SQL: {command.CommandText}");
    }

    partial void OnExecuted(string operationName, object? result, TimeSpan elapsed)
    {
        Console.WriteLine($"完成: {operationName}, 耗时: {elapsed.TotalMilliseconds}ms");
    }

    partial void OnExecuteFail(string operationName, Exception exception)
    {
        Console.WriteLine($"失败: {operationName}, 错误: {exception.Message}");
    }
}

{{set @param}} - 动态SET

// 动态指定更新字段
[Sqlx("UPDATE {{table}} SET {{set @updates}} WHERE id = @id")]
Task<int> UpdatePartialAsync(
    int id,
    [DynamicSql(Type = DynamicSqlType.Fragment)] string updates);

// 调用
await repo.UpdatePartialAsync(1, "name = @name, email = @email");
// 生成: UPDATE users SET name = @name, email = @email WHERE id = 1

适用场景： 部分字段更新、条件更新

{{orderby @param}} - 动态排序

// 动态排序规则
[Sqlx("SELECT {{columns}} FROM {{table}} {{orderby @sort}}")]
Task<List<Todo>> GetSortedAsync(
    [DynamicSql(Type = DynamicSqlType.Fragment)] string sort);

// 调用
await repo.GetSortedAsync("priority DESC, created_at DESC");
// 生成: SELECT * FROM todos ORDER BY priority DESC, created_at DESC

适用场景： 用户自定义排序、多列排序

{{join @param}} - 动态JOIN

// 动态JOIN子句
[Sqlx("SELECT o.*, c.name FROM orders o {{join @joins}} WHERE o.id = @id")]
Task<Order?> GetWithDetailsAsync(
    int id,
    [DynamicSql(Type = DynamicSqlType.Fragment)] string joins);

// 调用
await repo.GetWithDetailsAsync(1, "INNER JOIN customers c ON o.customer_id = c.id");
// 生成: SELECT o.*, c.name FROM orders o INNER JOIN customers c ON o.customer_id = c.id WHERE o.id = @id

适用场景： 复杂查询、可选关联

{{groupby @param}} - 动态GROUP BY

// 动态分组列
[Sqlx("SELECT {{groupby @cols}}, COUNT(*) FROM {{table}} {{groupby @cols}}")]
Task<List<Dictionary<string, object>>> GetAggregatedAsync(
    [DynamicSql(Type = DynamicSqlType.Fragment)] string cols);

// 调用
await repo.GetAggregatedAsync("category, status");
// 生成: SELECT category, status, COUNT(*) FROM items GROUP BY category, status

适用场景： 报表查询、聚合分析

🔒 安全保障

  • ✅ 所有动态占位符必须标记 [DynamicSql(Type = DynamicSqlType.Fragment)]
  • ✅ 自动包含 SqlValidator.IsValidFragment() 检查
  • ✅ 拒绝 DDL（DROP/TRUNCATE）、EXEC、注释等危险操作
  • ✅ 基于 ReadOnlySpan<char> 的零GC验证

// Sqlx 自动生成的方法
public async Task<List<Todo>> GetSortedAsync(string sort)
{
    // ✅ SQL片段验证（~50ns，零GC）
    if (!SqlValidator.IsValidFragment(sort.AsSpan()))
        throw new ArgumentException($"Invalid SQL fragment: {sort}. Contains dangerous keywords.", nameof(sort));

    var __orderByClause_0__ = sort;

    // ✅ 字符串插值（零Replace调用，编译时优化）
    __cmd__.CommandText = $@"SELECT id, title, priority FROM todos ORDER BY {__orderByClause_0__}";

    // ... 执行SQL
}
